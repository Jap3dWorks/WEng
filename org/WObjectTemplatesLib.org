#+title: WObject Templates Lib

* WObject hpp initial code

Template to create a new WObject class.

#+NAME: wobject-hpp-initial-code
#+begin_src elisp :var module="WEngineObjects" :var name="WObject" :var parent-module="" :var parent="WObject" :var namespace="weo" :var API="WENGINEOBJECTS"
(when (string-empty-p parent-module)
  (setq parent-module module))

(concat
 (format "#pragma once

#include \"WCore/WCore.hpp\"
#include \"WEngineObjects/WObjectMacros.hpp\"
#include \"%3$s/%4$s.hpp\"

#include \"%2$s.WEngine.hpp\"

namespace %5$s {

    class %6$s_API %2$s : public %4$s {

        WOBJECT_BODY;

    public:

    };

}" module name parent-module parent namespace API))
#+end_src

#+NAME: create-if-not-exists-wobject-hpp
#+HEADER: :var module="WEngineObjects" :var name="WObject" :var parent-module=""
#+HEADER: :var parent="WObject" :var namespace="weo" :var API="WENGINEOBJECTS"
#+begin_src elisp :noweb yes
(let ((file-path (format "Include/%s/%s.hpp" module name)))
(unless (f-exists-p file-path
  (with-temp-file file-path
    (insert <<wobject-hpp-initial-code(module, name, parent-module, parent, namespace, API)>>)))))
#+end_src
