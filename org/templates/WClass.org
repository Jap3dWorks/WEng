#+NAME: define-wclassname-macros
#+begin_src C++
#ifndef WENG_CLASSNAME
#define WENG_CLASSNAME(_CLS) WENG_DCONCAT(WClass__, _CLS)
#endif

#ifndef WENG_CLASSNAME_CLSS_VAR
#define WENG_CLASSNAME_CLSS_VAR(_CLS) WENG_DCONCAT(_VAR_, _CLS)
#endif

#ifndef WENG_CLASSNAME_STR
#define WENG_CLASSNAME_STR(_CLS) _STR(_CLS)
#endif

#ifndef WENG_CLASSNAME_STR_VAR
#define WENG_CLASSNAME_STR_VAR(_CLS) WENG_DCONCAT(_NAME_, _CLS)
#endif
#+end_src

** WClass.WEngine
:PROPERTIES:
:CUSTOM_ID: wclass-wengine
:END:

Content of wclass-wengine-hpp should be tangled inside PublicGenerated/ directory.

- Temporal WPROPERTY solution, more info check [[file:org/WEng.hpp::#WPROPERTY]]

#+NAME: wclass-wengine-hpp
#+HEADER: :var wclass="WObject" :var pwclass="WObject" :var weng-api="WENGINEOBJECTS"
#+HEADER: :var namespace="" :var pnamespace="" :noweb yes
#+begin_src elisp
(concat
 "
#pragma once
#include \"WClass/WClassFor.hpp\"
"
 "
#ifdef WCLASS
#undef WCLASS
#endif
#define WCLASS(...)
"
 "
#ifdef WPROPERTY
#undef WPROPERTY
#endif

#define WPROPERTY(_type, _name, _value)                                      \\
public:                                                                      \\
    inline _type Get_ ## _name () const { return _name ; };                   \\
    inline void Set_ ## _name (const _type & in_value) { _name = in_value; } \\
private:                                                                     \\
    _type _name { _value };
"
 (format
  "
#ifdef _WCLASS_
#undef _WCLASS_
#endif

#ifdef _PWCLASS_
#undef _PWCLASS_
#endif

#ifdef _WENG_API
#undef _WENG_API
#endif

#ifdef _WCLASS_NAMESPACE_
#undef _WCLASS_NAMESPACE_
#endif

#ifdef _PWCLASS_NAMESPACE_
#undef _PWCLASS_NAMESPACE_
#endif

#define _WCLASS_ %s
#define _PWCLASS_ %s
#define _WENG_API_ %s_API
#define _WCLASS_NAMESPACE_ %s
#define _PWCLASS_NAMESPACE_ %s
" wclass pwclass (upcase weng-api) (downcase namespace) (downcase pnamespace))

 "
#ifdef _WCLASS_DECLARATION_
#undef _WCLASS_DECLARATION_
#endif

#ifdef _PWCLASS_DECLARATION_
#undef _PWCLASS_DECLARATION_
#endif
"

 (format "
#define _WCLASS_DECLARATION_ %s
#define _PWCLASS_DECLARATION_ %s
"
         (if (string-empty-p namespace)
             "class _WCLASS_;"
           "namespace _WCLASS_NAMESPACE_ { class _WCLASS_; }")
         (if (string-empty-p pnamespace)
             "class _PWCLASS_;"
           "namespace _PWCLASS_NAMESPACE_ { class _PWCLASS_; }"
           ))

 "
<<define-dconcat>>
<<define-mp>>
<<define-wclassname-macros>>
"

 "
_WCLASS_DECLARATION_
_PWCLASS_DECLARATION_
"

 "
namespace wrf::wclass {
    inline constexpr std::string_view WENG_CLASSNAME_STR_VAR(WENG_MP( _WCLASS_ )) =
        WENG_CLASSNAME_STR(_WCLASS_);
"

"
    using WENG_CLASSNAME(_WCLASS_) = WClassFor<_WCLASS_NAMESPACE_ :: _WCLASS_, _PWCLASS_NAMESPACE_ :: _PWCLASS_>;
"

 "
    inline constexpr WENG_CLASSNAME(_WCLASS_) WENG_CLASSNAME_CLSS_VAR(_WCLASS_) =
        WENG_CLASSNAME_STR_VAR(WENG_MP( _WCLASS_ )) ;
}
"
 )
#+end_src
