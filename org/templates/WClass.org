#+NAME: wclass-macro-define-wjoin
#+begin_src C++
#ifndef IN_WJOIN1
#define IN_WJOIN1(A,B) A ## B
#endif

#ifndef IN_WJOIN2
#define IN_WJOIN2(A,B) IN_WJOIN1(A,B)
#endif

#ifndef WJOIN
#define WJOIN(A,B) IN_WJOIN2(A, B)
#endif
#+end_src
