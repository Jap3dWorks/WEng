** DECLARE_WCLASS
:PROPERTIES:
:CUSTOM_ID: declare_wclass
:END:

WClass es una clase que almacena datos relativos a otra clase clase derivada de WObject.

El archivo DECLARE_WCLASS define un WClass único para cada WObject.

#+NAME: declare_wclass
#+begin_src C++ :noweb yes :main no :eval no

#include "WCore/WCore.hpp"
#include "WCore/TSingleton.hpp"
#include "WReflection/WClassFor.hpp"

<<define-wclass-macros>>

<<define-dconcat>>

<<define-mp>>

<<define-wclassname-macros>>

_WCLASS_DECLARATION_
_PWCLASS_DECLARATION_

namespace wrf::wclass {
    inline constexpr std::string_view WENG_CLASSNAME_STR_VAR(WENG_MP( _WCLASS_ )) =
        WENG_CLASSNAME_STR(_WCLASS_);

    using WENG_CLASSNAME(_WCLASS_) = WClassFor<_WCLASS_, _PWCLASS_>;

    inline constexpr WENG_CLASSNAME(_WCLASS_) WENG_CLASSNAME_CLSS_VAR(_WCLASS_) =
        WENG_CLASSNAME_STR_VAR(WENG_MP( _WCLASS_ )) ;
}
#+end_src

** DEFINE_WCLASS
:PROPERTIES:
:CUSTOM_ID: define_wclass
:END:

El contenido de DEFINE_WCLASS debe ser compilado como un .cpp.

#+NAME: define_wclass
#+HEADER: :noweb yes :eval no :main no
#+begin_src C++
<<define-wclassname-macros>>

const WClass * _WCLASS_NAMESPACE_ :: _WCLASS_::StaticClass() noexcept {
    return & wrf::wclass::WENG_CLASSNAME_CLSS_VAR(_WCLASS_);
}
#+end_src

* PublicGenerated
** WClass.WEngine
:PROPERTIES:
:CUSTOM_ID: wclass-wengine
:END:

Content of wclass-wengine-hpp should be tangled inside PublicGenerated/ directory.

#+NAME: wclass-wengine-hpp
#+HEADER: :var wclass="WObject" :var pwclass="WObject" :var weng-api="WENGINEOBJECTS"
#+HEADER: :var nmespace=""
#+begin_src elisp
(concat
(format
"#pragma once

#ifdef _WCLASS_
#undef _WCLASS_
#endif

#ifdef _PWCLASS_
#undef _PWCLASS_
#endif

#ifdef _WENG_API
#undef _WENG_API
#endif

#ifdef _WCLASS_NAMESPACE_
#undef _WCLASS_NAMESPACE_
#endif

#ifdef _WCLASS_DECLARATION_
#undef _WCLASS_DECLARATION_
#endif

#ifdef _PWCLASS_DECLARATION_
#undef _PWCLASS_DECLARATION_
#endif

#ifdef WCLASS
#undef WCLASS
#endif

#define WCLASS(...)

#define _WCLASS_ %s
#define _PWCLASS_ %s
#define _WENG_API_ %s_API
#define _WCLASS_NAMESPACE_ %s

#include \"WReflection/_DECLARE_WCLASS_.inc\"
" wclass pwclass (upcase weng-api) nmespace))
#+end_src
