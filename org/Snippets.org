* Inherit Using A::A

#+HEADER: :flags --std=gnu++23
#+begin_src C++ :main no :results output :includes <iostream> <cstdint>

struct A {
    using Natural = std::uint32_t;
};

struct B : public A {
    using A::A;
};

int main() {

    B::Natural n = 3;

    std::cout << "Natural value: " << n << std::endl;

    return 0;
}

#+end_src

#+RESULTS:
: Natural value: 3
