(defun wng/find-files (root-dir regex)
  "Return list of files under ROOT-DIR whose names match REGEX.
ROOT-DIR is expanded relative to the current buffer's directory."
  (let* ((result '()))
    (dolist (file (directory-files-recursively root-dir "." t))
      (when (string-match-p regex (file-name-nondirectory file))
        (push file result)))
    (nreverse result)))
#+end_src

* Evaluate module org files

Evaluates and tangles each passed org file.

#+begin_src elisp :results value
(defun wng/evaluate-module-org-files (org-files)
  "Evaluate each .org file in ORG-FILES list.
Open each file, execute all source blocks and tangle, then close buffer."
  (dolist (file org-files)
    (when (string-match-p "\\.org$"  file)
      (ignore-errors
        (with-current-buffer (find-file-noselect file)
          (setq-local org-confirm-babel-evaluate nil)
          (print file)
          (org-babel-execute-buffer)
          (org-babel-tangle)
          (kill-current-buffer))))))
#+end_src

(defun wng/linkdef-single-module-string (module-name)
  "Returns a linkdef block for module-name."
  (concat
   (format
    "
#if defined(_WIN32)
    #ifdef %1$s_EXPORTS
        #define %1$s_API __declspec(dllexport)
    #else
        #define %1$s_API __declspec(dllimport)
    #endif
#elif defined(__GNUC__) || defined(__clang__)
    #ifdef %1$s_EXPORTS
        #define %1$s_API __attribute__((visibility(\"default\")))
    #else
        #define %1$s_API
    #endif
#else
    #define %1$s_API
#endif

"
    (upcase module-name)
    )
   )
  )
#+end_src

(defun wng/linkdef-complete-string (root-dir)
  (let* ((result '()))
    (dolist (file (wng/find-module-org-files root-dir))
      (push (wng/linkdef-single-module-string (file-name-base file)) result)
      )
    (apply #'concat result)
    )
  )
